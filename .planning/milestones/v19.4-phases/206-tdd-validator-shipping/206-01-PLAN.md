<feature>
  <name>TDD Validator Commands</name>
  <files>src/commands/misc/recovery.js</files>
  <tdd-targets>
    <red>node bgsd-tools.cjs execute:tdd validate-red --test-cmd "node test/fixture/calc.test.js" (must fail with missing behavior)</red>
    <green>node bgsd-tools.cjs execute:tdd validate-green --test-cmd "node test/fixture/calc.test.js" (must pass)</green>
    <refactor>node bgsd-tools.cjs execute:tdd validate-refactor --test-cmd "node test/fixture/calc.test.js" (must pass with same test count)</refactor>
  </tdd-targets>
  <behavior>
    Each validator command:
    1. Parses --test-cmd argument to extract the exact command to run
    2. Records test file mtime before running (for GREEN check)
    3. Runs command via spawnSync with utf-8 encoding
    4. Captures exit code, stdout, stderr
    5. For RED: validates exit !== 0 AND failure is about missing behavior (not crashes)
    6. For GREEN: validates exit === 0 AND test file mtime unchanged since record
    7. For REFACTOR: validates exit === 0 AND test count matches GREEN count
    8. Writes *-TDD-AUDIT.json sidecar atomically via writeFileAtomic
    9. Returns structured proof JSON with command, exit, evidence, stage, timestamp
  </behavior>
  <implementation>
    cmdTdd(cwd, subcommand, parsedArgs, raw) {
      switch(subcommand) {
        case 'validate-red':
          return validateRed(cwd, parsedArgs, raw);
        case 'validate-green':
          return validateGreen(cwd, parsedArgs, raw);
        case 'validate-refactor':
          return validateRefactor(cwd, parsedArgs, raw);
        case 'auto-test':
          return autoTest(cwd, parsedArgs, raw);
        default:
          output({ error: `Unknown subcommand: ${subcommand}` }, raw);
      }
    }
    
    validateRed: spawnSync test-cmd, verify exit !== 0, verify failure is semantic (not crash), write TDD-AUDIT.json, return proof JSON
    validateGreen: spawnSync test-cmd, verify exit === 0, verify test file unmodified, write TDD-AUDIT.json, return proof JSON
    validateRefactor: spawnSync test-cmd, verify exit === 0, verify test count unchanged from GREEN, write TDD-AUDIT.json, return proof JSON
  </implementation>
</feature>

    3. Add validateRed(cwd, args, raw):
       - Parse --test-cmd from args
       - Record test file mtime before running
       - Run command via spawnSync(testCmd, { cwd, encoding: 'utf-8', stdio: ['ignore', 'pipe', 'pipe'] })
       - Verify exit !== 0 (RED must fail)
       - Verify failure is semantic: stderr/stdout contains expected failure text, not crash/ENOENT
       - Build proof JSON: { stage: 'red', command: testCmd, exitCode, stdout, stderr, timestamp }
       - Write to *-TDD-AUDIT.json via writeFileAtomic
       - output(proof, raw)

    4. Add validateGreen(cwd, args, raw):
       - Parse --test-cmd from args
       - Parse --test-file from args (to track which file to verify unchanged)
       - Record test file mtime before running
       - Run command via spawnSync
       - Verify exit === 0 (GREEN must pass)
       - If --test-file provided, verify mtime unchanged (not modified during GREEN)
       - Build proof JSON with green-specific fields: { stage: 'green', command, exitCode, testFileUnmodified, timestamp }
       - Write TDD-AUDIT.json, output proof

    5. Add validateRefactor(cwd, args, raw):
       - Parse --test-cmd and --prev-count from args
       - Run command via spawnSync
       - Verify exit === 0
       - If --prev-count provided, verify test count unchanged
       - Build proof JSON: { stage: 'refactor', command, exitCode, testCount, timestamp }
       - Write TDD-AUDIT.json, output proof

    6. Add autoTest(cwd, args, raw):
       - Parse --test-cmd from args
       - Run command, return raw proof without validation
       - { stage: 'auto', command, exitCode, stdout, stderr, timestamp }

    7. Update cmdTdd switch to call these functions:
       switch(subcommand) {
