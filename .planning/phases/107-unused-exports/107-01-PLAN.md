# Create inventory script and run it
1. Create a Node.js script that uses src/lib/ast.js extractExports() function
2. Scan all .js files in src/ directory recursively
3. For each file, extract: named exports, default export, CJS exports
4. Output to .planning/phases/107-unused-exports/exports-inventory.json

Script should:
- Use require('./src/lib/ast') to get extractExports
- Walk src/ directory for all .js files
- Call extractExports() on each file
- Collect results into array: {file, named[], default, cjsExports[], type}
- Write JSON output
  </action>
  <verify>
node -e "
const fs = require('fs');
const path = require('path');
const { extractExports } = require('./src/lib/ast');

const srcDir = './src';
const exports = [];

function walk(dir) {
  const files = fs.readdirSync(dir);
  for (const file of files) {
    const fullPath = path.join(dir, file);
    const stat = fs.statSync(fullPath);
    if (stat.isDirectory()) {
      walk(fullPath);
    } else if (file.endsWith('.js')) {
      const result = extractExports(fullPath);
      if (result.named.length > 0 || result.default || result.cjsExports.length > 0) {
        exports.push({
          file: fullPath,
          named: result.named,
          default: result.default,
          cjsExports: result.cjsExports,
          type: result.type
        });
      }
    }
  }
}

walk(srcDir);
console.log('Total files with exports:', exports.length);
console.log('Sample:', JSON.stringify(exports[0], null, 2));
"
</verify>
